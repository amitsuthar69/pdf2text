PDF to Text Conversion API

Convert PDF files to plain text through a simple REST API

Base URL

https://pdf2text-pixl3aibmq-el.a.run.app/convert

Endpoint

POST /convert

Request

Headers

Key Value Description
Content-Type multipart/form-data Required for file upload

Body

form-data: pdf (File)

The PDF file to be processed

Response

{
  "text": "Extracted text from PDF..."
}

Example

cURL

curl -X POST "https://pdf2text-pixl3aibmq-el.a.run.app/convert" \
     -H "Content-Type: multipart/form-data" \
     -F "pdf=@file.pdf"

JavaScript

const fs = require("fs");
const axios = require("axios");
const FormData = require("form-data");

async function extractTextFromPDF(filePath) {
  const API_URL = "https://pdf2text-pixl3aibmq-el.a.run.app/convert";
  const fileStream = fs.createReadStream(filePath);

  const formData = new FormData();
  formData.append("pdf", fileStream);

  try {
    // Sending a POST request to our deployed endpoint.
    const response = await axios.post(API_URL, formData);
    return response.data.text;
  } catch (error) {
    console.error("Error extracting text from PDF:", error);
    return null;
  }
}

extractTextFromPDF("./my-pdf.pdf").then((extractedText) => {
  extractedText
    ? console.log("Extracted Text: \n", extractedText)
    : "Unable to extract Text!";
});

Python

import requests
from pathlib import Path

async def extract_text_from_pdf(file_path):
  API_URL = "https://pdf2text-pixl3aibmq-el.a.run.app/convert"
  
  try:
    with open(file_path, 'rb') as file:
      file_stream = file.read()
      files = {'pdf': file_stream}
      response = requests.post(API_URL, files=files)
      if response.status_code == 200:
        return response.json().get('text')
      else:
        print(f"API request failed with status code: {response.status_code}")
        return None
              
  except Exception as e:
    print(f"Error extracting text from PDF: {str(e)}")
    return None
